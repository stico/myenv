	Inbox
		Wine uses a virtual C: drive in your home directory's .wine/drive_c subdirectory
		if env var WINEPREFIX is set, wine uses it to find the .wine directory.

		If need run internet based application like a browser in WINE, you need to have a package called "Wine Gecko".

		even num version: stable

# Var
wine_ver=1.6

wine_src_dir=$MY_PRO/wine_`date "+%Y%m%d_%H%M%S"`_src

apt_src=/etc/apt/sources.list
os_code=`cat /etc/lsb-release | sed -n -e "s/DISTRIB_CODENAME=//p"`

# Install dep, add apt src and ppa
[ ! -e ${apt_src}.bak.forWineOnLM ] && cp ${apt_src}{,.bak.forWineOnLM}
echo "deb-src http://archive.ubuntu.com/ubuntu/ ${os_code} main restricted universe multiverse" >> $apt_src
sudo add-apt-repository ppa:ubuntu-wine/ppa
sudo apt-get update
sudo apt-get build-dep wine${wine_ver}-amd64
